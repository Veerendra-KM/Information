Comprehensive SQL Commands Guide

This document provides an end-to-end overview of SQL commands using practical business-oriented examples.

1. Data Retrieval Commands

SELECT

SELECT employee_name FROM employees;

SELECT *

SELECT * FROM products;

SELECT DISTINCT

SELECT DISTINCT department FROM employees;

SELECT INTO

SELECT * INTO archived_sales FROM sales_2023;

SELECT TOP / LIMIT

SELECT TOP 5 * FROM orders; -- MySQL alternative: SELECT * FROM orders LIMIT 5;

2. Filtering Data

WHERE

SELECT product_name FROM products WHERE price > 1000;

AND / OR

SELECT emp_name FROM employees WHERE department = 'IT' AND experience > 3;

BETWEEN

SELECT student_name FROM students WHERE marks BETWEEN 60 AND 85;

LIKE

SELECT customer_name FROM customers WHERE customer_name LIKE 'A%';

IN

SELECT city FROM suppliers WHERE city IN ('Delhi', 'Pune', 'Bengaluru');

IS NULL / IS NOT NULL

SELECT emp_name FROM employees WHERE manager_id IS NULL;

3. Data Definition Language (DDL)

CREATE DATABASE

CREATE DATABASE enterpriseDB;

CREATE TABLE

CREATE TABLE employees ( emp_id INT PRIMARY KEY, emp_name VARCHAR(100), department VARCHAR(50), salary INT );

ALTER TABLE

ALTER TABLE employees ADD email VARCHAR(100);

DROP TABLE

DROP TABLE temporary_records;

TRUNCATE TABLE

TRUNCATE TABLE logs;

4. Data Manipulation Language (DML)

INSERT

INSERT INTO employees VALUES (101, 'Rahul', 'Finance', 70000);

UPDATE

UPDATE employees SET salary = 75000 WHERE emp_id = 101;

DELETE

DELETE FROM employees WHERE emp_id = 120;

5. Aggregate Functions

SELECT COUNT(*) FROM orders; SELECT AVG(salary) FROM employees; SELECT MAX(marks), MIN(marks) FROM students; SELECT department, SUM(salary) FROM employees GROUP BY department;

6. Grouping and Sorting

GROUP BY & HAVING

SELECT department, COUNT(emp_id) FROM employees GROUP BY department HAVING COUNT(emp_id) > 5;

ORDER BY

SELECT product_name FROM products ORDER BY price DESC;

7. Joins

INNER JOIN

SELECT o.order_id, c.customer_name FROM orders o INNER JOIN customers c ON o.customer_id = c.customer_id;

LEFT JOIN

SELECT e.emp_name, p.project_name FROM employees e LEFT JOIN projects p ON e.emp_id = p.emp_id;

FULL JOIN

SELECT * FROM students FULL OUTER JOIN results ON students.roll_no = results.roll_no;

8. Advanced SQL Commands

CASE

SELECT emp_name, CASE WHEN salary > 80000 THEN 'High' ELSE 'Medium' END AS salary_level FROM employees;

Subquery

SELECT emp_name FROM employees WHERE salary > (SELECT AVG(salary) FROM employees);

Common Table Expression (CTE)

WITH DeptAvg AS ( SELECT department, AVG(salary) avg_salary FROM employees GROUP BY department ) SELECT * FROM DeptAvg;

Window Function

SELECT emp_name, salary, RANK() OVER (ORDER BY salary DESC) AS rank_salary FROM employees;

EXISTS

SELECT customer_name FROM customers c WHERE EXISTS ( SELECT 1 FROM orders o WHERE o.customer_id = c.customer_id );

9. Transaction Control Commands

BEGIN TRANSACTION; UPDATE accounts SET balance = balance - 500 WHERE acc_id = 1; SAVEPOINT deduct_done; ROLLBACK TO deduct_done; COMMIT;

10. Access Control Commands

GRANT

GRANT SELECT, INSERT ON employees TO user_hr;

REVOKE

REVOKE INSERT ON employees FROM user_hr;
